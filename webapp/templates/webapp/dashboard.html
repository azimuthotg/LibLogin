{% if not has_default_bg %}
ไม่มี Default Background — ปัจจุบันทุก background ผูกกับ hotspot เฉพาะ การเรียก API โดยไม่ใส่ ?hotspot_name= จะได้ 404 อัปโหลด background แบบ Default
{% endif %}
1
Base URL ของ API
{{ request.scheme }}://{{ request.get_host }}{% url 'login-background' %}
2
ต้องใส่ ?hotspot_name= เสมอ
ระบุชื่อ hotspot ให้ตรงกับที่ตั้งค่าไว้ในระบบ เพื่อดึง background ของ router นั้น ถ้าไม่ใส่และไม่มี Default background จะได้ 404
{{ request.scheme }}://{{ request.get_host }}{% url 'login-background' %}?hotspot_name=hotspot_wifi
3
ใส่ใน MikroTik login.html
ในไฟล์ login.html ของ MikroTik ใช้ fetch() เรียก API แล้วนำ imageUrl ไปใส่เป็น background
// ตัวอย่าง JavaScript ใน MikroTik login.html fetch('{{ request.scheme }}://{{ request.get_host }}{% url 'login-background' %}?hotspot_name=hotspot_wifi') .then(r => r.json()) .then(data => { if (data.success) { document.body.style.backgroundImage = `url(${data.imageUrl})`; } });
